#!/usr/bin/env python3
"""
migrate_codeberg_to_github.py

Idempotent migration of labels, milestones, and issues (with comments)
from a Codeberg (Gitea) repo to a GitHub repo.

Usage:
  CODEBERG_TOKEN=xxx GITHUB_TOKEN=yyy python tools/migrate_codeberg_to_github.py \
    --codeberg-owner foxprimegames --codeberg-repo climatomaton \
    --github-owner YOUR_GH_ORG --github-repo YOUR_GH_REPO
"""

import requests
import time
import argparse
import sys

CODEBERG_API = "https://codeberg.org/api/v1"
GITHUB_API = "https://api.github.com"
MARKER_TEMPLATE = "<!-- CODEBERG-ISSUE-ID: {id} -->"

# Simple retry helper
def retry_request(fn, max_tries=3, sleep=1):
    for i in range(max_tries):
        r = fn()
        if r is not None and r.status_code < 500:
            return r
        time.sleep(sleep)
    return r

def paginate_get(session, url, params=None, headers=None):
    params = params or {}
    page = 1
    per_page = 50
    results = []
    while True:
        p = params.copy()
        p.update({"page": page, "limit": per_page})
        r = retry_request(lambda: session.get(url, params=p, headers=headers))
        if r is None:
            raise RuntimeError(f"Failed GET {url}")
        if r.status_code != 200:
            raise RuntimeError(f"GET {url} failed: {r.status_code} {r.text}")
        batch = r.json()
        if not batch:
            break
        results.extend(batch)
        if len(batch) < per_page:
            break
        page += 1
    return results

def paginate_get_github(session, url, params=None):
    params = params or {}
    page = 1
    per_page = 100
    results = []
    headers = {}
    while True:
        p = params.copy()
        p.update({"page": page, "per_page": per_page})
        r = retry_request(lambda: session.get(url, params=p, headers=headers))
        if r is None:
            raise RuntimeError(f"Failed GET {url}")
        if r.status_code != 200:
            raise RuntimeError(f"GET {url} failed: {r.status_code} {r.text}")
        batch = r.json()
        if not batch:
            break
        results.extend(batch)
        if len(batch) < per_page:
            break
        page += 1
    return results

def ensure_label_on_github(gh_sess, gh_owner, gh_repo, label):
    # label: dict with name, color, description
    name = label.get("name")
    url = f"{GITHUB_API}/repos/{gh_owner}/{gh_repo}/labels/{requests.utils.quote(name, safe='')}"
    r = gh_sess.get(url)
    if r.status_code == 200:
        return
    # create
    payload = {"name": name, "color": label.get("color", "ffffff")}
    if label.get("description"):
        payload["description"] = label["description"]
    r = gh_sess.post(f"{GITHUB_API}/repos/{gh_owner}/{gh_repo}/labels", json=payload)
    if r.status_code not in (200, 201):
        raise RuntimeError(f"Failed to create label {name}: {r.status_code} {r.text}")

def ensure_milestone_on_github(gh_sess, gh_owner, gh_repo, milestone):
    # milestone: dict with title, description, due_on
    title = milestone.get("title")
    # get existing milestones
    existing = paginate_get_github(gh_sess, f"{GITHUB_API}/repos/{gh_owner}/{gh_repo}/milestones", params={"state":"all"})
    for m in existing:
        if m.get("title") == title:
            return
    payload = {"title": title, "description": milestone.get("description") or ""}
    if milestone.get("due_on"):
        payload["due_on"] = milestone["due_on"]
    r = gh_sess.post(f"{GITHUB_API}/repos/{gh_owner}/{gh_repo}/milestones", json=payload)
    if r.status_code not in (200, 201):
        raise RuntimeError(f"Failed to create milestone {title}: {r.status_code} {r.text}")

def build_issue_import_object(cb_issue, cb_comments):
    # cb_issue: codeberg issue dict
    # cb_comments: list of comment dicts
    body = cb_issue.get("body") or ""
    # append marker for idempotency
    marker = MARKER_TEMPLATE.format(id=cb_issue["number"])
    body_with_marker = f"{body}\n\n{marker}"
    issue_obj = {
        "title": cb_issue.get("title") or f"Issue {cb_issue['number']}",
        "body": body_with_marker,
        "created_at": cb_issue.get("created_at"),
        "closed": (cb_issue.get("state") == "closed"),
        "labels": [l["name"] for l in (cb_issue.get("labels") or [])],
        "milestone": cb_issue.get("milestone", {}).get("title") if cb_issue.get("milestone") else None,
        "comments": []
    }
    for c in cb_comments:
        comment_obj = {
            "created_at": c.get("created_at"),
            "body": c.get("body") or "",
            "author": c.get("user", {}).get("login") or None
        }
        issue_obj["comments"].append(comment_obj)
    if cb_issue.get("closed_at"):
        issue_obj["closed_at"] = cb_issue["closed_at"]
    return issue_obj

def main():
    p = argparse.ArgumentParser()
    p.add_argument("--codeberg-api", default=CODEBERG_API)
    p.add_argument("--codeberg-owner", required=True)
    p.add_argument("--codeberg-repo", required=True)
    p.add_argument("--github-owner", required=True)
    p.add_argument("--github-repo", required=True)
    p.add_argument("--batch-size", type=int, default=50, help="How many issues to send per import batch")
    args = p.parse_args()

    CODEBERG_TOKEN = os_envar("CODEBERG_TOKEN")
    GITHUB_TOKEN = os_envar("GITHUB_TOKEN")
    cb_base = args.codeberg_api.rstrip("/")
    cb_owner = args.codeberg_owner
    cb_repo = args.codeberg_repo
    gh_owner = args.github_owner
    gh_repo = args.github_repo

    # sessions
    cb_sess = requests.Session()
    cb_sess.headers.update({"Authorization": f"token {CODEBERG_TOKEN}"})

    gh_sess = requests.Session()
    gh_sess.headers.update({
        "Authorization": f"token {GITHUB_TOKEN}",
        "Accept": "application/vnd.github.v3+json"
    })

    # 1) migrate labels
    print("Fetching Codeberg labels...")
    cb_labels = paginate_get(cb_sess, f"{cb_base}/repos/{cb_owner}/{cb_repo}/labels")
    print(f"Found {len(cb_labels)} labels on Codeberg. Ensuring on GitHub...")
    for lb in cb_labels:
        l = {"name": lb.get("name"), "color": lb.get("color", "ffffff"), "description": lb.get("description") or ""}
        ensure_label_on_github(gh_sess, gh_owner, gh_repo, l)
    print("Labels synced.")

    # 2) migrate milestones
    print("Fetching Codeberg milestones...")
    cb_milestones = paginate_get(cb_sess, f"{cb_base}/repos/{cb_owner}/{cb_repo}/milestones", params={"state":"all"})
    print(f"Found {len(cb_milestones)} milestones on Codeberg. Ensuring on GitHub...")
    for ms in cb_milestones:
        ensure_milestone_on_github(gh_sess, gh_owner, gh_repo, ms)
    print("Milestones synced.")

    # 3) build set of already-imported Codeberg issue ids on GitHub
    print("Scanning GitHub issues for existing Codeberg markers (idempotency)...")
    existing_github_issues = paginate_get_github(gh_sess, f"{GITHUB_API}/repos/{gh_owner}/{gh_repo}/issues", params={"state":"all"})
    existing_markers = set()
    for gi in existing_github_issues:
        body = gi.get("body") or ""
        # find marker
        if "<!-- CODEBERG-ISSUE-ID:" in body:
            # crude extraction
            import re
            m = re.search(r"<!-- CODEBERG-ISSUE-ID: ([0-9]+) -->", body)
            if m:
                existing_markers.add(int(m.group(1)))
    print(f"Found {len(existing_markers)} already-migrated Codeberg issues on GitHub.")

    # 4) fetch Codeberg issues and prepare import batch
    print("Fetching Codeberg issues...")
    cb_issues = paginate_get(cb_sess, f"{cb_base}/repos/{cb_owner}/{cb_repo}/issues", params={"state":"all"})
    print(f"Found {len(cb_issues)} Codeberg issues total.")
    to_import = []
    for issue in cb_issues:
        # note: Codeberg issue number is issue['number']
        if issue["number"] in existing_markers:
            continue
        # fetch comments
        comments = paginate_get(cb_sess, f"{cb_base}/repos/{cb_owner}/{cb_repo}/issues/{issue['number']}/comments")
        issue_obj = build_issue_import_object(issue, comments)
        to_import.append(issue_obj)

    if not to_import:
        print("No new issues to import. Exiting.")
        return

    # 5) import issues in batches using GitHub Issue Import API
    print(f"Preparing to import {len(to_import)} issues to GitHub in batches of {args.batch_size}...")
    import_endpoint = f"{GITHUB_API}/repos/{gh_owner}/{gh_repo}/import/issues"
    for i in range(0, len(to_import), args.batch_size):
        batch = to_import[i:i+args.batch_size]
        payload = {"issues": batch}
        r = gh_sess.post(import_endpoint, json=payload)
        if r.status_code not in (201, 202):
            print(f"Import request failed: {r.status_code} {r.text}")
            sys.exit(1)
        # GitHub returns a url to poll, in location or in response json 'url'
        status_url = r.headers.get("Location") or r.json().get("url")
        if not status_url:
            print("No status URL returned for import; aborting.")
            sys.exit(1)
        # poll until done
        print(f"Imported batch starting at {i}. Polling status at {status_url} ...")
        while True:
            s = retry_request(lambda: gh_sess.get(status_url))
            if s is None:
                print("Status poll failed. Exiting.")
                sys.exit(1)
            status = s.json()
            state = status.get("status")
            if state in ("imported", "complete"):
                print("Batch import completed.")
                break
            if state == "failed":
                print("Import failed:", status)
                sys.exit(1)
            # else pending
            time.sleep(2)

    print("All requested issues imported. Note: GitHub will create issues and preserve timestamps/comments via the Import API.")
    print("Done.")

def os_envar(name):
    import os
    v = os.environ.get(name)
    if not v:
        print(f"Environment variable {name} is required.")
        sys.exit(1)
    return v

if __name__ == "__main__":
    main()
